strmqm  TESTQMGR1

3. Test if it started successfully

ps -ef|grep mqm

4. Now go into TESTQMGR1 to configure it

runmqsc TESTQMGR1

5.For options avaiablble

DISPLAY

6. To display QUEUES avaiablble. It displays even system queues

DISPLAY QLOCAL(*)

7. Create new local queue "Q1LOCAL"

DEFINE QLOCAL (Q1LOCAL)

8. Create Listener 

DEFINE LISTENER(TESTQMGR1.LISTENER) TRPTYPE(TCP) PORT(30001) IPADDR(192.168.2.151)

9. To dsiplay listeners

DISPLAY LISTENER(*)

10. To start the listener you created

START LISTENER(TESTQMGR1.LISTENER)

11. To check if listener started

ps -ef |grep mqlsr or ps -ef|grep mqm

12. Telnet from other system to 30001 if not able to telnet open port

firewall-cmd --zone=public --add-port=30001/tcp --permanent
firewall-cmd --reload

13. Alter ore replace new admin.svrconn channel

DEFINE CHANNEL(SYSTEM.ADMIN.SVRCONN) CHLTYPE(SVRCONN) TRPTYPE(TCP) DESCR('WMQ DEFAULT CHANNEL') REPLACE

	DISPLAY  QLOCAL(SQM_TQ1)
	ALTER  QLOCAL(SQM_TQ1) NOTRIGGER


--> ALTER QL(<Trans Q Name>) TRIGGER TRIGTYPE (<FIRST/DEPTH/EVERY>) TRIGDATA(<SENDER CHNL NAME>) INITQ (SYSTEM.CHANNEL.INITQ) (for first)
---> ALTER QL(<Trans Q Name>) TRIGGER TRIGTYPE (DEPTH) TRIGDPTH (<Number>)TRIGDATA(<SENDER CHNL NAME>) INITQ (SYSTEM.CHANNEL.INITQ) (for depth)

ALTER QL(SQM_TQ1) TRIGGER TRIGTYPE (FIRST) TRIGDATA(SQM_RQM_CH1) INITQ (SYSTEM.CHANNEL.INITQ)
ALTER QL(SQM_TQ1) TRIGGER TRIGTYPE (DEPTH) TRIGDPTH(5) TRIGDATA(SQM_RQM_CH1) INITQ (SYSTEM.CHANNEL.INITQ)

# To inactivate channel
STOP CHL(SQM_RQM_CH1) STATUS(INACTIVE)

# To chnage disconnect interval-- in seconds -- need restart of channel
ALTER CHL(SQM_RQM_CH1) CHLTYPE(SDR) DISCINT(20)
